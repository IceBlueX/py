CREATE DATABASE test
  WITH OWNER = libaokun
       ENCODING = 'UTF8'
       TABLESPACE = pg_default
       LC_COLLATE = 'zh_CN.UTF-8'
       LC_CTYPE = 'zh_CN.UTF-8'
       CONNECTION LIMIT = -1;




-- Table: public.blogs

-- DROP TABLE public.blogs;

CREATE TABLE public.blogs
(
  id character varying(50) NOT NULL, -- not null
  user_id character varying(50) NOT NULL,
  user_name character varying(50) NOT NULL,
  user_image character varying(500) NOT NULL,
  name character varying(50) NOT NULL,
  summary character varying(200) NOT NULL,
  content text NOT NULL,
  created_at real NOT NULL,
  CONSTRAINT blogs_pkey PRIMARY KEY (id),
  CONSTRAINT blogs_created_at_key UNIQUE (created_at)
)
WITH (
  OIDS=FALSE
);
ALTER TABLE public.blogs
  OWNER TO libaokun;
COMMENT ON COLUMN public.blogs.id IS 'not null';

(

  OIDS=FALSE

  OWNER TO libaokun;
